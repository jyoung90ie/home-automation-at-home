249 def try_to_link_zigbee_device(self) -> None: 

250 """Filters ZigbeeDevice objects with device friendly_name and device_identifier to see if 

251 there are any unlinked matches - which will be then linked to the current device.""" 

252 if self.protocol != DeviceProtocol.ZIGBEE or self.get_zigbee_device(): 

253 logger.debug("Device already linked") 

254 return 

255 

256 if not self.zigbee_model: 

257 logger.debug( 

258 "Cannot link zigbee device - ZigbeeDevice model has not been setup in __init__()" 

259 ) 

260 return 

261 

262 try: 

263 logger.info("Attempting to link device...") 

264 # not using Q model here as priority is on a hierarchical match - if using 

265 # Q, potential for two matches exists 

266 zigbee_device = self.zigbee_model.objects.filter( 

267 Q(ieee_address=self.device_identifier) 

268 | Q(friendly_name=self.friendly_name) 

269 ) 

270 

271 if zigbee_device and not hasattr(zigbee_device, "device"): 

272 # obj found and is not already matched 

273 zigbee_device = zigbee_device.first() 

274 

275 zigbee_device.device = self 

276 zigbee_device.save() 

277 

278 logger.info("Device linked - ZigbeeDevice obj= %s", zigbee_device) 

279 

280 except (self.zigbee_model.DoesNotExist, Exception) as ex: 

281 logger.error("Device could not be linked - %s", ex)
